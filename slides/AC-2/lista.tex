\textsc{\textbf{Lista de Exercícios}}

{\bf Cálculo da Complexidade}

\end{center}

\vspace{0.2in}

\begin{enumerate}

	\item Proponha um algoritmo para resolver os problemas abaixo e determine a complexidade assintótica de cada um deles:

	\begin{enumerate}
		\item Encontrar a média dos elementos de um vetor de inteiros $a$ de tamanho $n$.

		\item Garantir que uma string que represente um parágrafo começe em letra maiúscula.

		\item Multiplicar duas matrizes quadradas de ordem $n$.

	\end{enumerate}

	\item Proponha um algoritmo para resolver os problemas abaixo e determine o melhor caso, o pior caso e o caso médio de cada um deles:

	\begin{enumerate}
		\item Encontrar o maior elemento de um vetor de inteiros $a$ 
		de tamanho $n$.

		\item Dados dois vetores $\vec{v}_1$ e $\vec{v}_2$, encontrar 
		um par de vetores ortogonais $\vec{u}_1$ e $\vec{u}_2$ 
		({\it Sugestão:} Utilizar a ortogonalização de Gram-Schmidt).

	\end{enumerate}

\end{enumerate}
